package t_translator;

import junit.framework.TestCase;

public class StartTTest extends TestCase {

protected StartT ST;
protected StartT.SetUpTCode tProgram;

@Override
protected void setUp() throws Exception {
  super.setUp();
  ST = new StartT();
  tProgram = (StartT.SetUpTCode)ST.startNewTProgram();
}

@Override protected void 
tearDown() throws Exception {
  tProgram = null;
  ST = null;
  super.tearDown();
}

/******** TESTS ********/

public void 
testResultOfParse() {
  tProgram.programCodeLines("class OrderedPair {");
  tProgram.programCodeLines("private int first;");
  tProgram.programCodeLines("private int second;");
  tProgram.programCodeLines("");
  tProgram.programCodeLines("void swap() {");
  tProgram.programCodeLines("  javalanguage {int temp = first;}");
  tProgram.programCodeLines("  conjecture .999 {temp' = 'first;}");
  tProgram.programCodeLines("}");
  tProgram.programCodeLines("} // end class OrderedPair");
  
  final String resultOfParse = ST.resultOfParse();
  assertEquals("OK", resultOfParse);
}

public void 
testJavaOutput() {
  assertEquals("OK","OK");
}

}
